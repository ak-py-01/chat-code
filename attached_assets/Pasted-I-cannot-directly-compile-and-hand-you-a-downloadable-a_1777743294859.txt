### 3. `app/src/main/java/com/dashcam/app/ocr/SpeedLimitDetector.kt`
*This handles the ML Kit logic to find speed limits without cloud APIs.*
```kotlin
package com.dashcam.app.ocr

import android.graphics.Rect
import com.google.mlkit.vision.common.InputImage
import com.google.mlkit.vision.text.TextRecognition
import com.google.mlkit.vision.text.latin.TextRecognizerOptions

class SpeedLimitDetector {

    private val recognizer = TextRecognition.getClient(TextRecognizerOptions.DEFAULT_OPTIONS)
    private val validLimits = listOf(20, 30, 40, 50, 60, 70, 80, 90, 100, 110, 120, 130)

    interface DetectionListener {
        fun onSpeedLimitDetected(limit: Int)
        fun onNoLimitDetected()
    }

    fun analyzeFrame(imageProxy: android.media.ImageProxy, listener: DetectionListener) {
        val mediaImage = imageProxy.image
        if (mediaImage != null) {
            val image = InputImage.fromMediaImage(mediaImage, imageProxy.imageInfo.rotationDegrees)
            
            recognizer.process(image)
                .addOnSuccessListener { visionText ->
                    var bestMatch: Int? = null
                    var highestConfidenceScore = 0

                    for (block in visionText.textBlocks) {
                        val text = block.text.uppercase()
                        val boundingBox = block.boundingBox

                        // Strategy 1: Look for explicit "SPEED LIMIT X" or "X KM/H"
                        val explicitMatch = Regex("""(\d{2,3})\s*(?:KM\/H|MPH|KPH)""").find(text)
                            ?: Regex("""SPEED\s*LIMIT\s*(\d{2,3})""").find(text)
                            ?: Regex("""LIMIT\s*(\d{2,3})""").find(text)

                        if (explicitMatch != null) {
                            val num = explicitMatch.groupValues[1].toIntOrNull()
                            if (num != null && num in validLimits) {
                                bestMatch = num
                                break // High confidence, stop searching
                            }
                        }

                        // Strategy 2: Isolated numbers (often all that's visible on a sign)
                        val isolatedMatch = Regex("""\b(\d{2,3})\b""").find(text)
                        if (isolatedMatch != null) {
                            val num = isolatedMatch.groupValues[1].toIntOrNull()
                            if (num != null && num in validLimits) {
                                // Heuristic: Signs are usually in the upper half of the frame
                                // and relatively square-ish or tall. Give higher score to upper regions.
                                val score = if (boundingBox.top < (imageProxy.height / 2)) 10 else 5
                                if (score > highestConfidenceScore) {
                                    highestConfidenceScore = score
                                    bestMatch = num
                                }
                            }
                        }
                    }

                    if (bestMatch != null) {
                        listener.onSpeedLimitDetected(bestMatch)
                    } else {
                        listener.onNoLimitDetected()
                    }
                }
                .addOnFailureListener {
                    listener.onNoLimitDetected()
                }
                .addOnCompleteListener {
                    imageProxy.close() // CRITICAL: Must close to receive next frame
                }
        } else {
            imageProxy.close()
        }
    }
}
```

*Handles long polling and the specific commands requested.*
```kotlin
package com.dashcam.app.telegram

import android.content.Context
import android.util.Log
import kotlinx.coroutines.*
import org.json.JSONObject
import java.io.File
import okhttp3.*

class TelegramBotClient(private val context: Context) {

    private val client = OkHttpClient()
    private var pollingJob: Job? = null
    private var lastUpdateId = 0

    fun startPolling(token: String, chatId: String, commandHandler: TelegramCommandHandler) {
        if (pollingJob?.isActive == true) return
        
        pollingJob = CoroutineScope(Dispatchers.IO).launch {
            while (isActive) {
                try {
                    val url = "https://api.telegram.org/bot$token/getUpdates?offset=${lastUpdateId + 1}&timeout=10"
                    val request = Request.Builder().url(url).build()
                    val response = client.newCall(request).execute()
                    val json = JSONObject(response.body?.string() ?: "{}")
                    val results = json.optJSONArray("ok")?.let { json.getJSONArray("result") } ?: json.getJSONArray("result")

                    for (i in 0 until results.length()) {
                        val update = results.getJSONObject(i)
                        lastUpdateId = update.getInt("update_id")
                        val message = update.optJSONObject("message") ?: continue
                        val text = message.optString("text", "")
                        
                        // Only respond to the authorized chat ID
                        if (message.getJSONObject("chat").getString("id") != chatId) continue

                        when {
                            text == "/instant" -> commandHandler.handleInstant(chatId, token)
                            text == "/today" -> commandHandler.handleToday(chatId, token)
                            text.startsWith("/date ") -> {
                                val dateStr = text.removePrefix("/date ").trim()
                                commandHandler.handleDate(chatId, token, dateStr)
                            }
                        }
                    }
                } catch (e: Exception) {
                    Log.e("TelegramBot", "Polling error: ${e.message}")
                    delay(5000) // Back off on error
                }
            }
        }
    }

    fun stopPolling() {
        pollingJob?.cancel()
    }

    // Helper to send text messages
    fun sendMessage(token: String, chatId: String, text: String) {
        CoroutineScope(Dispatchers.IO).launch {
            try {
                val url = "https://api.telegram.org/bot$token/sendMessage"
                val jsonBody = JSONObject().apply {
                    put("chat_id", chatId)
                    put("text", text)
                    put("parse_mode", "HTML")
                }
                val requestBody = RequestBody.create(MediaType.parse("application/json"), jsonBody.toString())
                val request = Request.Builder().url(url).post(requestBody).build()
                client.newCall(request).execute()
            } catch (e: Exception) {
                Log.e("TelegramBot", "Send failed: ${e.message}")
            }
        }
    }

    // Helper to send documents (text files)
    fun sendDocument(token: String, chatId: String, file: File, caption: String) {
        CoroutineScope(Dispatchers.IO).launch {
            try {
                val url = "https://api.telegram.org/bot$token/sendDocument"
                val requestBody = MultipartBody.Builder()
                    .setType(MultipartBody.FORM)
                    .addFormDataPart("chat_id", chatId)
                    .addFormDataPart("caption", caption)
                    .addFormDataPart("document", file.name, RequestBody.create(MediaType.parse("text/plain"), file))
                    .build()
                val request = Request.Builder().url(url).post(requestBody).build()
                client.newCall(request).execute()
            } catch (e: Exception) {
                Log.e("TelegramBot", "Doc send failed: ${e.message}")
            }
        }
    }
}
```

```kotlin
    private lateinit var locationManager: LocationManager
    private var currentSpeedKmh: Float = 0f
    private var detectedSpeedLimit: Int? = null
    private var lastBeepTime = 0L
    
    private lateinit var telegramClient: TelegramBotClient

    private fun checkOverspeed() {
        val limit = detectedSpeedLimit ?: return
        if (currentSpeedKmh > limit) {
            val now = System.currentTimeMillis()
            if (now - lastBeepTime > 15000) { // 15 second cooldown
                playBeep()
                lastBeepTime = now
            }
            // TODO: Broadcast to UI to turn speed text RED
        }

    // Programmatic beep generation (Fallback if raw resource is missing)
    private fun playBeep() {
        serviceScope.launch(Dispatchers.IO) {
            try {
                val sampleRate = 44100
                val duration = 0.2 // seconds
                val numSamples = (sampleRate * duration).toInt()
                val beepBuffer = ShortArray(numSamples)
                val frequency = 1200.0
                
                for (i in 0 until numSamples) {
                    val sineWave = Math.sin(2 * Math.PI * frequency * i / sampleRate)
                    beepBuffer[i] = (sineWave * Short.MAX_VALUE * 0.5).toInt().toShort()
                }

                val audioTrack = AudioTrack.Builder()
                    .setAudioAttributes(AudioAttributes.Builder().setUsage(AudioAttributes.USAGE_ALARM).setContentType(AudioAttributes.CONTENT_TYPE_SONIFICATION).build())
                    .setAudioFormat(android.media.AudioFormat.Builder().setEncoding(android.media.AudioFormat.ENCODING_PCM_16BIT).setSampleRate(sampleRate).setChannelMask(android.media.AudioFormat.CHANNEL_OUT_MONO).build())
                    .setBufferSizeInBytes(numSamples * 2)
                    .build()
                
                audioTrack.write(beepBuffer, 0, numSamples)
                audioTrack.play()
                Thread.sleep(200)
                audioTrack.stop()
                audioTrack.release()
            } catch (e: Exception) { Log.e("DashCam", "Beep failed") }
        }
    }

    private fun stopRecording() {
        locationManager.removeUpdates(this)
        telegramClient.stopPolling()
        serviceScope.cancel()
    }

    private fun createNotificationChannel() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            val channel = NotificationChannel(CHANNEL_ID, "DashCam Service", NotificationManager.IMPORTANCE_LOW)
            val manager = getSystemService(NotificationManager::class.java)
            manager.createNotificationChannel(channel)
        }
    }

    private fun buildNotification(text: String): Notification {
        return NotificationCompat.Builder(this, CHANNEL_ID)
            .setContentTitle("DashCam")
            .setContentText(text)
            .setSmallIcon(R.drawable.ic_notification) // Ensure you have this icon in res/drawable
            .build()
    }

    private fun updateNotification(text: String) {
        val manager = getSystemService(NotificationManager::class.java)
        manager.notify(NOTIFICATION_ID, buildNotification(text))
    }

    override fun onDestroy() {
        super.onDestroy()
        stopRecording()
    }

    override fun onBind(intent: Intent?): IBinder? = null
```
